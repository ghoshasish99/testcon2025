import com.microsoft.playwright.*;
import com.microsoft.playwright.options.*;
import static com.microsoft.playwright.assertions.PlaywrightAssertions.assertThat;
import java.util.*;

public class Example {
  public static void main(String[] args) {
    try (Playwright playwright = Playwright.create()) {
      Browser browser = playwright.chromium().launch(new BrowserType.LaunchOptions()
        .setHeadless(false));
      BrowserContext context = browser.newContext();
      Page page = context.newPage();
      page.navigate("https://ecommerce-playground.lambdatest.io/");
      page.getByRole(AriaRole.BUTTON, new Page.GetByRoleOptions().setName("ï€‡ My account")).click();
      page.getByRole(AriaRole.LINK, new Page.GetByRoleOptions().setName("Register").setExact(true)).click();
      page.getByRole(AriaRole.TEXTBOX, new Page.GetByRoleOptions().setName("First Name*")).click();
      page.getByRole(AriaRole.TEXTBOX, new Page.GetByRoleOptions().setName("First Name*")).fill("TestAccount1");
      page.getByRole(AriaRole.TEXTBOX, new Page.GetByRoleOptions().setName("Last Name*")).click();
      page.getByRole(AriaRole.TEXTBOX, new Page.GetByRoleOptions().setName("Last Name*")).fill("TestAccountLastName1");
      page.getByRole(AriaRole.TEXTBOX, new Page.GetByRoleOptions().setName("E-Mail*")).click();
      page.getByRole(AriaRole.TEXTBOX, new Page.GetByRoleOptions().setName("E-Mail*")).fill("testaccount12345@gmail.com");
      page.getByRole(AriaRole.TEXTBOX, new Page.GetByRoleOptions().setName("Telephone*")).click();
      page.getByRole(AriaRole.TEXTBOX, new Page.GetByRoleOptions().setName("Telephone*")).fill("678123467");
      page.getByRole(AriaRole.TEXTBOX, new Page.GetByRoleOptions().setName("Password*")).click();
      page.getByRole(AriaRole.TEXTBOX, new Page.GetByRoleOptions().setName("Password*")).fill("test123");
      page.getByRole(AriaRole.TEXTBOX, new Page.GetByRoleOptions().setName("Password Confirm*")).click();
      page.getByRole(AriaRole.TEXTBOX, new Page.GetByRoleOptions().setName("Password Confirm*")).fill("test123");
      page.getByText("I have read and agree to the").click();
      page.getByRole(AriaRole.BUTTON, new Page.GetByRoleOptions().setName("Continue")).click();
      assertThat(page.locator("h1")).containsText("Your Account Has Been Created!");
    }
  }
}
